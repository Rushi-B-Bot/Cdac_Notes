package com.Student;

import java.util.Scanner;

public class Student_GetData1 {

	public Student_GetData1(int n) {
		 Studnet_DisplayData   S1[ ]= new Studnet_DisplayData [n];
		 Scanner obj = new Scanner(System.in);
		 for(int i=0;i<n;i++)
		 {
			 System.out.println("Enter Student of id " + (i + 1) + ":");
	            int id = obj.nextInt();
	            System.out.println("Enter Student of name " + (i + 1) + ":");
	            String name = obj.next();
	            System.out.println("Enter Student Addmisson Date " + (i + 1) + ":");
	            String Date = obj.next();
	            
	            S1[i]=new Studnet_DisplayData(id,name,Date);	 
		 }
		 
		 System.out.println("enter Student id to be search");
			int pid=obj.nextInt();
			Studnet_DisplayData p=Student_GetData1.searchPerson(S1,pid);
			if(p!=null) {
				System.out.println(p);
			}else {
				System.out.println("not found");
			}
		 

	}

	private static Studnet_DisplayData searchPerson(Studnet_DisplayData[] s1, int pid) {
	
		for(Studnet_DisplayData p:s1) {   
			if(p.getId()==pid) {
				return p;
			}
		}
		return null;
	} 
}
